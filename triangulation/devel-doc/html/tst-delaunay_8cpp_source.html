00049   std::ifstream ifstr;

00069       int itmp;

00072         nbpts = itmp;
00073       else
00074         if(itmp != 0) throw("read_ply: unexpected non nul faces nb\n");
00075     } else if(name == "property") {
00076       std::string str2;
00077       ifstr >> str2;
00078       if(str1 == "float") nbflt++;
00079       else if (str1 == "uchar") nbint++;
00080       else if(str1 == "list") {
00081         ifstr.getline(tmp, 256);
00082         continue;
00083       }
00084       else throw("read_ply: unexpected property type\n");
00085     } else {
00086       std::cout << "ERR " << name << std::endl;
00087       throw("read_ply: unexpected keyword\n");
00088     }
00089     if (ifstr.eof()) throw("read_ply: unexpected EOF\n");
00090   }
00091   if((nbflt != 3 && nbflt != 6) || (nbint != 0 && nbint != 3)) 
00092     throw("read_ply: unexpected data structure\n");
00093   int j = points.size();
00094   for(int i = 0;i < nbpts;i++) {
00095     if (ifstr.eof()) {
00096       std::cout << "I = " << i << std::endl;
00097       throw("read_ply: points reading unexpected EOF\n");
00098     }
00099     ifstr >> x >> y >> z;
00100     if(nbflt > 3)
00101       ifstr >> nx >>ny >>nz;
00102     if(nbint > 0)
00103       ifstr >> r >> g >> b;
00104     points.push_back( std::make_pair(Point(x,y,z),j++));
00105   }
00106   ifstr.close();
